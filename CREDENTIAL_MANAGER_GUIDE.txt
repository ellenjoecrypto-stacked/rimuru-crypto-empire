Integration Example (rimuru_ai.py):

```python
from credential_manager_api import ProgrammaticCredentialManager

class RimuruAICore:
    def __init__(self, master_password: str):
        self.cred_mgr = ProgrammaticCredentialManager(
            master_password=master_password
        )
    
    def get_exchange_credentials(self, exchange: str):
        """Get API credentials for exchange"""
        api_key = self.cred_mgr.get(f"{exchange} API Key")
        api_secret = self.cred_mgr.get(f"{exchange} API Secret")
        return api_key, api_secret
    
    def rotate_credentials(self, exchange: str, new_key: str):
        """Rotate API key for exchange"""
        self.cred_mgr.update(f"{exchange} API Key", new_key)
```
